# Detect the operating system
OS="$(uname)"

# Function to determine if sudo is needed
needs_sudo() {
    if [ "$OS" = "Linux" ]; then
        echo "sudo"
    else
        echo ""
    fi
}

# CakePHP Shells
alias cake_shell='$(needs_sudo) docker compose exec php bin/cake "$@"'
alias php_exec='$(needs_sudo) docker compose exec php "$@"'
alias php_shell='$(needs_sudo) docker compose exec -it php /bin/sh'

# Testing Commands

# Change ownership of current directory to current user
change_ownership() {
    local current_user=$(whoami)
    local current_group=$(id -gn)
    $(needs_sudo) chown -R "$current_user:$current_group" .
    echo "Ownership set to $current_user:$current_group for current directory and subdirectories."
}
